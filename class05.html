#2A. Line plot
weight <- read.table("bimm143_05_rstats/weight_chart.txt", header=TRUE)

plot(weight$Age,weight$Weight,
     typ="o",
     pch = 15, cex = 1.5, lwd = 2,
     ylim = c(2,10),
     xlab = "Age (months)", ylab = "Weight (kg)", main = "Baby weight with age")

#2B. Barplot
#sep: separates data entries
#header = TRUE means there's a header
feat <- read.table("bimm143_05_rstats/feature_counts.txt",
           sep="\t", header=TRUE)

par(mar = c(5.1,12,4.1,10)) #mar sets margins (bottom, left, top, right)
barplot(feat$Count,names.arg = feat$Feature, horiz = TRUE,
        xlab = "Count", main = "Number of features in the mouse GRCm38 gene",
        xlim = c(0,80000),
        las = 1)

#par() function is used to edit graph parameters
#"las" is the style of axis labels (1 is horizontal, 2 is vertical etc.)
par()$mar
## [1]  5.1 12.0  4.1 10.0
